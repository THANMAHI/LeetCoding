class Solution {
    public int earliestTime(int[][] tasks) {
        int max = Integer.MAX_VALUE;
        for(int i=0;i<tasks.length;i++){
            int si = tasks[i][0];
            int ti = tasks[i][1];
            int total = si+ti;

            if(total<max){
                max = total;
            }
        }
        return max;
    }
}
